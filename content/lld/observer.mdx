## Concept

The **Observer** is a behavioral design pattern that lets you define a subscription mechanism to notify multiple objects about any events that happen to the object they're observing.

Often known as "Pub-Sub" (Publisher-Subscriber), it's heavily used in Event-Driven systems.

## When to use

- When changes to the state of one object may require changing other objects, and the actual set of objects is unknown beforehand or changes dynamically.
- When you want to decouple the core component (Publisher) from the dependent components (Subscribers) so that adding new subscribers doesn't require modifying the publisher.
- Extremely common in UI frameworks (e.g., a Button click event notifying multiple listener functions).

interface Observer {
  update(temperature: number): void;
}

// 2. Subject Interface (Publisher)
interface Subject {
  attach(observer: Observer): void;
  detach(observer: Observer): void;
  notify(): void;
}

// 3. Concrete Subject
class WeatherStation implements Subject {
  private observers: Observer[] = [];
  private temperature: number = 0;

  attach(observer: Observer): void {
    const isExist = this.observers.includes(observer);
    if (!isExist) this.observers.push(observer);
  }

  detach(observer: Observer): void {
    const observerIndex = this.observers.indexOf(observer);
    if (observerIndex !== -1) {
      this.observers.splice(observerIndex, 1);
    }
  }

  notify(): void {
    console.log("WeatherStation: Notifying observers...");
    for (const observer of this.observers) {
      observer.update(this.temperature);
    }
  }

  setTemperature(temp: number): void {
    console.log(\`\nWeatherStation: Temperature changed to \${temp}\`);
    this.temperature = temp;
    this.notify();
  }
}

// 4. Concrete Observers (Subscribers)
class PhoneDisplay implements Observer {
  update(temperature: number): void {
    console.log(\`Phone: Alert! Temperature is now \${temperature} degrees\`);
  }
}

class TVDisplay implements Observer {
  update(temperature: number): void {
    console.log(\`TV Banner: Current temp: \${temperature}\`);
  }
}

// 5. Client Code
const weatherStation = new WeatherStation();
const phone = new PhoneDisplay();
const tv = new TVDisplay();

weatherStation.attach(phone);
weatherStation.attach(tv);

weatherStation.setTemperature(25);
weatherStation.detach(tv);
weatherStation.setTemperature(30);`}

interface Observer {
  void update(int temperature);

interface Subject {
  void attach(Observer o);
  void detach(Observer o);
  void notifyObservers();
}

  private int temperature;

  public void attach(Observer o) { observers.add(o); }
  public void detach(Observer o) { observers.remove(o); }

    this.temperature = temp;
    notifyObservers();
  }

  public void notifyObservers() {
    for (Observer o : observers) {
      o.update(temperature);
    }
  }
}

    Observer phone = new PhoneDisplay();
    Observer tv = new TVDisplay();

}`}

# Observer
class Observer(ABC):
    @abstractmethod
    def update(self, temperature: int) -> None: pass

# Subject
class Subject(ABC):
    @abstractmethod
    def attach(self, observer: Observer) -> None: pass
    @abstractmethod
    def detach(self, observer: Observer) -> None: pass
    @abstractmethod
    def notify(self) -> None: pass

class WeatherStation(Subject):
    def __init__(self):
        self._observers = []
        self._temperature = 0

    def attach(self, observer: Observer) -> None:
        if observer not in self._observers:
            self._observers.append(observer)

    def detach(self, observer: Observer) -> None:
        self._observers.remove(observer)

    def set_temperature(self, temp: int) -> None:
        self._temperature = temp
        self.notify()

    def notify(self) -> None:
        for observer in self._observers:
            observer.update(self._temperature)

## Gotchas

- **Memory Leaks (The Lapsed Listener Problem)**: If an observer registers with a subject, but the observer is later discarded by the client without calling `detach()`, the subject still holds a strong reference to the observer in its array. This prevents the garbage collector from freeing the memory, causing a classic memory leak in long-running apps.
- **Update Cascades / Cycles**: Be extremely careful if observers in turn update other subjects. This can cause infinite loops or massive chain-reaction updates (Event Cascades) that cripple performance.
