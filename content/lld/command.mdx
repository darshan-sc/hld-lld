## Concept

The **Command** is a behavioral design pattern that turns a request into a stand-alone object that contains all information about the request.

This transformation lets you pass requests as a method argument, delay or queue a request's execution, and support undoable operations.

## When to use

- When you want to parameterize objects with operations (e.g., configuring GUI buttons to execute different actions depending on the context).
- When you want to queue operations, schedule their execution, or execute them remotely.
- When you want to implement reversible operations (Undo/Redo). You keep a history of executed command objects and call an `undo()` method on them.

class Light {
  turnOn(): void { console.log("Light is ON"); }
  turnOff(): void { console.log("Light is OFF"); }
}

// 2. The Command Interface
interface Command {
  execute(): void;
  undo(): void;
}

// 3. Concrete Commands encapsulate requests and delegate to Receiver
class LightOnCommand implements Command {
  constructor(private light: Light) {}
  
  execute(): void { this.light.turnOn(); }
  undo(): void { this.light.turnOff(); }
}

class LightOffCommand implements Command {
  constructor(private light: Light) {}

  execute(): void { this.light.turnOff(); }
  undo(): void { this.light.turnOn(); }
}

// 4. The Invoker (Executes commands, can optionally keep history)
class RemoteControl {
  private commandHistory: Command[] = [];

  executeCommand(command: Command): void {
    command.execute();
    this.commandHistory.push(command);
  }

  undoLastCommand(): void {
    const last = this.commandHistory.pop();
    if (last) {
      console.log("Undoing last action...");
      last.undo();
    }
  }
}

// 5. Client Code
const livingRoomLight = new Light();
const remote = new RemoteControl();

const turnOn = new LightOnCommand(livingRoomLight);
const turnOff = new LightOffCommand(livingRoomLight);

remote.executeCommand(turnOn);
remote.executeCommand(turnOff);
remote.undoLastCommand(); // Undoes the 'turnOff', turns light back ON`}

  public void turnOn() { System.out.println("Light ON"); }
  public void turnOff() { System.out.println("Light OFF"); }

interface Command {
  void execute();
  void undo();
}

// Concrete Commands
class LightOnCommand implements Command {
  private Light light;
  public LightOnCommand(Light light) { this.light = light; }
  
  public void execute() { light.turnOn(); }
  public void undo() { light.turnOff(); }

class LightOffCommand implements Command {
  private Light light;
  public LightOffCommand(Light light) { this.light = light; }
  
  public void execute() { light.turnOff(); }
  public void undo() { light.turnOn(); }
}

// Invoker
import java.util.Stack;
class RemoteControl {
  private Stack<Command> history = new Stack<>();
  
  public void submit(Command c) {
    c.execute();
    history.push(c);
  }
  
  public void undo() {
    if (!history.isEmpty()) {
      history.pop().undo();
    }
  }
}

public class Main {
  public static void main(String[] args) {
    Light light = new Light();
    RemoteControl remote = new RemoteControl();
    
    remote.submit(new LightOnCommand(light));
    remote.submit(new LightOffCommand(light));
    remote.undo();
  }

# Receiver
class Light:
    def turn_on(self): print("Light is ON")
    def turn_off(self): print("Light is OFF")

# Command Interface
class Command(ABC):
    @abstractmethod
    def execute(self) -> None: pass
    @abstractmethod
    def undo(self) -> None: pass

# Concrete Commands
class LightOnCommand(Command):
    def __init__(self, light: Light): self.light = light
    def execute(self) -> None: self.light.turn_on()
    def undo(self) -> None: self.light.turn_off()

class LightOffCommand(Command):

    def execute(self) -> None: self.light.turn_off()
    def undo(self) -> None: self.light.turn_on()

    def submit(self, command: Command):
        command.execute()
        self.history.append(command)

    def undo(self):
        if self.history:
            self.history.pop().undo()

## Gotchas

- **Too many classes**: Every single unique action requires creating a brand new Command subclass. If you have hundreds of actions in a GUI, you'll have hundreds of classes.
- **Command vs Strategy**: Both encapsulate a behavior, but their intents differ. Command encapsulates a specific *request* with all its parameters (e.g., "Delete File X"). Strategy encapsulates an *algorithm* to solve a problem (e.g., "Sort using MergeSort" or "Sort using QuickSort").
